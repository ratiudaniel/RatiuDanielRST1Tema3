    <label for="text">Introdu textul:</label>

    <input type="text" id="text"><br><br>

    <button onclick="capitalizare()">Capitalizare</button><br><br>

    <p id="rezultat"></p>

    <script>

      function capitalizare() {

        var text = document.getElementById("text").value;

        var cuvinte = text.split(" ");

        for (var i = 0; i < cuvinte.length; i++) {

          cuvinte[i] = cuvinte[i].charAt(0).toUpperCase() + cuvinte[i].slice(1);

        }

        var rezultat = cuvinte.join(" ");

        document.getElementById("rezultat").innerHTML = "Textul capitalizat este: " + rezultat;

      }

    </script>
